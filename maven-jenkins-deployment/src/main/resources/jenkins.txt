pipeline {
agent any
stages {
stage(‘Build’) {
steps {
sh ‘mvn clean install’
}
}
stage(‘Deploy’) {
environment {
ANYPOINT_CREDENTIALS = credentials(‘anypointPlatform’)
}
steps {
bat ‘mvn clean deploy -DmuleDeploy -Duser.clientId=${ANYPOINT_CREDENTIALS_USR} -Duser.clientSecret=${ANYPOINT_CREDENTIALS_PSW} -DworkerType=Micro -Dworkers=1 -Dregion=us-east-1’
}
}
}
}
